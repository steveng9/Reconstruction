New system design

results.db (SQLite, experiment_scripts/results.db):

```sql
-- One row per unique experiment run
CREATE TABLE runs (
    run_id       INTEGER PRIMARY KEY AUTOINCREMENT,
    dataset      TEXT    NOT NULL,   -- 'adult', 'nist_sbo', etc.
    dataset_size INTEGER NOT NULL,   -- 1000, 10000, 100000
    sample       INTEGER NOT NULL,   -- 0–9
    qi           TEXT    NOT NULL,   -- 'QI1', 'QI_large', etc.
    sdg_method   TEXT    NOT NULL,   -- 'MST_eps1', 'RankSwap', etc.
    attack       TEXT    NOT NULL,   -- 'RandomForest', 'MarginalRF_mst_local_100', etc.
    split        TEXT    NOT NULL DEFAULT 'standard',  -- 'standard', 'train', 'nontraining'
    ra_mean      REAL,
    attack_params_json TEXT,
    sdg_params_json    TEXT,
    source_file        TEXT,
    wandb_run_id       TEXT,
    wandb_group        TEXT,
    ingested_at        TEXT,         -- ISO timestamp
    confidence         TEXT NOT NULL DEFAULT 'certain',
    confidence_notes   TEXT,
    UNIQUE(dataset, dataset_size, sample, qi, sdg_method, attack, split)
);

-- One row per feature score
CREATE TABLE feature_scores (
    score_id  INTEGER PRIMARY KEY AUTOINCREMENT,
    run_id    INTEGER NOT NULL REFERENCES runs(run_id),
    feature   TEXT    NOT NULL,
    ra_score  REAL,
    UNIQUE(run_id, feature)
);

-- Duplicate conflicts — never overwrite, always document
CREATE TABLE conflicts (
    conflict_id      INTEGER PRIMARY KEY AUTOINCREMENT,
    existing_run_id  INTEGER REFERENCES runs(run_id),
    new_source_file  TEXT,
    new_wandb_run_id TEXT,
    new_ra_mean      REAL,
    logged_at        TEXT
);
```
